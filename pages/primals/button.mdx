import { Button } from '@/primals';

# Button

The `Button` component is a styled button element that can be used to trigger actions in your application.

```jsx
<Button
  size="(sm|lg)"
  variant="(success|info|warning|danger)"
  onClick={() => alert('Button clicked!')}
>
  Click me!
</Button>
```

## Props

| Prop     | Type                                         | Description                                     | Default |
|:---------|:---------------------------------------------|:------------------------------------------------|:--------|
| children | ReactNode                                    | The content of the Button component.            | -       |
| height   | number \| string                             | The height of the Button container.             | -       |
| m        | number \| string                             | The margin around the Button container.         | -       |
| mb       | number \| string                             | The bottom margin of the Button container.      | -       |
| ml       | number \| string                             | The left margin of the Button container.        | -       |
| mr       | number \| string                             | The right margin of the Button container.       | -       |
| mt       | number \| string                             | The top margin of the Button container.         | -       |
| mx       | number \| string                             | The horizontal margin of the Button container.  | -       |
| my       | number \| string                             | The vertical margin of the Button container.    | -       |
| p        | number \| string                             | The padding around the Button container.        | 2       |
| pb       | number \| string                             | The bottom padding of the Button container.     | -       |
| pl       | number \| string                             | The left padding of the Button container.       | -       |
| pr       | number \| string                             | The right padding of the Button container.      | -       |
| pt       | number \| string                             | The top padding of the Button container.        | -       |
| px       | number \| string                             | The horizontal padding of the Button container. | -       |
| py       | number \| string                             | The vertical padding of the Button container.   | -       |
| width    | number \| string                             | The width of the Button container.              | -       |
| variant  | "danger" \| "info" \| "success" \| "warning" | The variant of the Button component.            | -       |

## Examples

### Sizes

<Button size="sm">Small</Button>

<Button size="md">Medium</Button>

<Button size="lg">Large</Button>

### Variants

<Button variant="success" transition='unset'>Success</Button>

<Button variant="info">Info</Button>

<Button variant="warning">Warning</Button>

<Button variant="danger">Danger</Button>
